#include <stdio.h>
#include <string.h>

int main()
{
    char str[25], str2[25];

    printf("Enter String 1: ");
    fgets(str, sizeof(str), stdin);

    printf("Enter String 2: ");
    fgets(str2, sizeof(str2), stdin);

    if(strcmp(str, str2) == 0)
        printf("Both strings are equal");
    else
        printf("Both strings are not equal");

    return 0;
}
