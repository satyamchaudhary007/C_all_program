#include <stdio.h>

int main()
{
    char str[100];
    int i = 0, length;

    printf("Enter the string: ");
    fgets(str, sizeof(str), stdin);

    while(str[i] != '\0')
    {
        i++;
    }

    length = i;

    printf("The length of string is: %d", length);

    return 0;
}
